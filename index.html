    <style>

        #map { height: 600px; width: 100%; }

        .info {

            padding: 6px 8px;

            font: 14px/16px Arial, Helvetica, sans-serif;

            background: white;

            background: rgba(255,255,255,0.8);

            box-shadow: 0 0 15px rgba(0,0,0,0.2);

        }

<body>

    <div id="map"></div>


    <script src="https://unpkg.com/leaflet/dist/leaflet.js"></script>

    <script>

        // Sample GeoJSON data containing total contributions: unweighted and weighted means

        var statesData = {

            "type": "FeatureCollection",

            "features": [

                {

                    "type": "Feature",

                    "properties": {

                        "name": "California",

                        "unweighted_mean": 11.26,

                        "weighted_mean": 11.79

                    },

                    "geometry": {

                        "type": "Polygon",

                        "coordinates": [

                            // Add California coordinates here

                        ]

                    }

                },

                {

                    "type": "Feature",

                    "properties": {

                        "name": "Texas",

                        "unweighted_mean": 4.28,

                        "weighted_mean": 4.21

                    },

                    "geometry": {

                        "type": "Polygon",

                        "coordinates": [

                            // Add Texas coordinates here

                        ]

                    }

                }

                // Add more states here...

            ]

        };


        // Initialize the map

        var map = L.map('map').setView([37.8, -96], 4); // Centered on the US


        // Tile layer (base map)

        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

            attribution: '&copy; OpenStreetMap contributors'

        }).addTo(map);


        // Define the Hoover Institution color scheme (example)

        function getColor(d) {

            return d > 15 ? '#800026' :

                   d > 10 ? '#BD0026' :

                   d > 7  ? '#E31A1C' :

                   d > 5  ? '#FC4E2A' :

                   d > 3  ? '#FD8D3C' :

                   d > 1  ? '#FEB24C' :

                            '#FFEDA0';

        }


        // Styling function for each state

        function style(feature) {

            return {

                fillColor: getColor(feature.properties.weighted_mean), // Color based on weighted mean

                weight: 2,

                opacity: 1,

                color: 'white',

                dashArray: '3',

                fillOpacity: 0.7

            };

        }


        // Add GeoJSON layer and style it based on the weighted mean

        L.geoJson(statesData, {

            style: style,

            onEachFeature: onEachFeature // Add interactivity

        }).addTo(map);


        // Adding interactivity: Display info on hover

        function highlightFeature(e) {

            var layer = e.target;


            layer.setStyle({

                weight: 5,

                color: '#666',

                dashArray: '',

                fillOpacity: 0.7

            });


            info.update(layer.feature.properties); // Update info box with state data

        }


        function resetHighlight(e) {

            geojson.resetStyle(e.target); // Reset to original style

            info.update(); // Clear info box

        }


        // Define what happens on hover

        function onEachFeature(feature, layer) {

            layer.on({

                mouseover: highlightFeature,

                mouseout: resetHighlight

            });

        }


        // Control that shows state info on hover

        var info = L.control();


        info.onAdd = function (map) {

            this._div = L.DomUtil.create('div', 'info'); // Create a div with class 'info'

            this.update();

            return this._div;

        };


        // Method to update the control based on feature properties passed

        info.update = function (props) {

            this._div.innerHTML = '<h4>Total Contributions</h4>' +  (props ?

                '<b>' + props.name + '</b><br />' + props.unweighted_mean + ' (Unweighted Mean)<br />' +

                props.weighted_mean + ' (Weighted Mean)' : 'Hover over a state');

        };


        info.addTo(map); // Add the info box to the map

    </script>
